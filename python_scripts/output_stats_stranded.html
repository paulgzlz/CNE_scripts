# Data dictionaries

cne_counts = {}
cne_similarity = {}
cne_lengths = {}
# Keep track of species already seen
seen_species = []
# Read each output file
for filepath in cne_files:
    file = filepath.split("/")[-1]
    #filepath = cnefinder_out_dir + file
    print("Processing file: ", filepath)
    # Sometimes, CNEFinder makes output file even when no CNEs are found
    if os.stat(filepath).st_size == 0:
        print("File exmpty, let's skip this file")
        continue
    # Retrieve species names
    species_pair = file.split(".")[0].split("_vs_")
    ref_species = species_pair[0]
    query_species = species_pair[1]
    # Read CNEFinder output file
    cne_df = pd.read_csv(filepath, sep="\t")
                         #names=["ref_scaff", "ref_start", "ref_end", "ref_strand", 
                          #      "query_scaff", "query_end", "query_strand", "ref_length", "query_length", "similarity"])
    # Retrieve CNE count
    cne_count = len(cne_df)
    for species in species_pair:
        if species in seen_species:
            cne_counts[species] += cne_count
        else:
            cne_counts[species] = cne_count 
    
    # Read each row and retrieve CNE lengths and similarity
    for idx, row in cne_df.iterrows():
        ref_length = row['ref_length']
        query_length = row['query_length']
        similarity = row['similarity']
        if ref_species in seen_species:
            cne_lengths[ref_species].append(ref_length)
            cne_similarity[ref_species].append(similarity)
        else:
            cne_lengths[ref_species] = [ref_length]
            cne_similarity[ref_species] = [similarity]
        if query_species in seen_species:
            cne_lengths[query_species].append(query_length)
            cne_similarity[query_species].append(similarity)
        else:
            cne_lengths[query_species] = [query_length]
            cne_similarity[query_species] = [similarity]
    # Update list of seen species before moving to the next file
    seen_species = list(set(seen_species + species_pair))
